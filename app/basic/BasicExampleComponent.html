<div class="panel panel-default">
    <div class="panel-heading">Cities population and areas</div>

    <table class="table table-striped" [mfData]="data" #mf="mfDataTable">
        <thead>
        <tr>
            <th style="width: 20%">
                <mfDefaultSorter by="country">Country</mfDefaultSorter>
            </th>
            <th style="width: 50%">
                <mfDefaultSorter by="city">City</mfDefaultSorter>
            </th>
            <th style="width: 10%">
                <mfDefaultSorter by="area">Area</mfDefaultSorter>
            </th>
            <th style="width: 20%">
                <mfDefaultSorter by="population">Population</mfDefaultSorter>
            </th>
        </tr>
        </thead>
        <tbody>
        <tr *ngFor="#item of mf.data">
            <td>{{item.country | uppercase}}</td>
            <td>{{item.city}}</td>
            <td class="text-right">{{item.area}}</td>
            <td class="text-right">{{item.population}}</td>
        </tr>
        </tbody>

        <tr>
            <td colspan="4">
                <mfBootstrapPaginator [activePage]="5" [rowsOnPageSet]="[5,10,25]" [rowsOnPage]="5"></mfBootstrapPaginator>
            </td>
        </tr>
